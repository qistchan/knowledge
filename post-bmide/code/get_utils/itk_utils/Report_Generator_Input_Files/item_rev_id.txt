QUERY.HEADER=ITEM.ITEM_ID~ITEMREVISION.ITEM_REVISION_ID~ITEM.TYPE
QUERY.OUTPUTTYPES=string,string,string
QUERY.SQL=select distinct i.pitem_id, ir.pitem_revision_id, wso.pobject_type from pitemrevision ir, pitem i, pworkspaceobject wso where (( not REGEXP_LIKE (ir.pitem_revision_id, '[A-HJ-NPR-Z]{1,2}') or REGEXP_LIKE (ir.pitem_revision_id, '.{3,}') and not REGEXP_LIKE (ir.pitem_revision_id, '[0-9A-Z]{1,2}\.[0-9]{3}')) and ir.pitem_revision_id!='0' and ir.pitem_revision_id!='00' and ir.pitem_revision_id!='000' and not REGEXP_LIKE (ir.pitem_revision_id, '[0-9A-Z]{1,2}\.[0-9]{3}') and ir.ritems_tagu=i.puid and wso.puid=i.puid and upper(wso.pobject_type) in ('GET6MECHPART','GET6SOFTWAREPART','GET6DRAWING','GET6ELECTPART','GET6DOCUMENT','GET6SPECS','GET6PROJECT','GET6MODULE','GET6WORKAREA','GET6FLATPATTERN','GET6MATERIAL')) or ((not REGEXP_LIKE (ir.pitem_revision_id, '[0-9]{3}') or REGEXP_LIKE (ir.pitem_revision_id, '.{4,}')) and not REGEXP_LIKE (ir.pitem_revision_id, '[0-9A-Z]{1,2}\.[0-9]{3}') and ir.ritems_tagu=i.puid and wso.puid=i.puid and upper(wso.pobject_type) in ('GET6MEDEPARTMENT','GET6MFGPART','GET6OPERATION','GET6PROCESS','GET6STATION','GET6MACHINE','GET6OPERATOR'))
